pipelines:
  branches:
    stage:   #Branch name
      - step:
          name: Sync stage Branch to GitHub
          image: alpine/git:latest
          clone:
            enabled: false
          script:
            - git clone -b stage --single-branch git@bitbucket.org:alliondev/allion-saas.git allion-saas.git    #Bitbucket Repository ssh url <space> Repository name
            - cd allion-saas.git   #Bitbucket Repository name
            - git push git@github.com:damith47/bitbucket-sync.git stage  #Github Repository ssh url
